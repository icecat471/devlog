html 캔버스에 이미지 그리기

April 14, 2021 · 1 min · icecat471 
<canvas id="myCanvas" width="500" height="800"></canvas>

let canvas = document.querySelector('#myCanvas');
let context = canvas.getContext('2d');

// 이미지 로딩
let dragon = new Image();
dragon.src = '이미지 경로';

context.drawImage(dragon, x좌표, y좌표, width, height);

setInterval()을 사용해서 특정시간마다 화면을 새로 그릴 수 있다.

// 1초마다 새로 그림
setInterval(()=> {
    context.drawImage(dragon, x좌표, y좌표, width, height);
}, 1000); // 1000ms
